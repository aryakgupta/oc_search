  54.     /**
  55.      * Constructor. Takes the raw HTTP response body and the exploded HTTP headers
  56.      *
  57.      * @param string $rawResponse 
  58.      * @param array $httpHeaders 
  59.      */
  60.     public function __construct($rawResponse, $httpHeaders = array())
  61.     {
  62.         //Assume 0, 'Communication Error', utf-8, and  text/xml
  63.         $status = 0;
  64.         $statusMessage = 'Communication Error';
  65.         $type = 'text/plain';
  66.         $encoding = 'UTF-8';
  67.  
  68.         //iterate through headers for real status, type, and encoding
  69.         if (is_array($httpHeaders) && count($httpHeaders) > 0)
  70.         {
  71.             //look at the first header for the HTTP status code
  72.             //and message (errors are usually returned this way)
  73.             if (substr($httpHeaders[0], 0, 4) == 'HTTP')
  74.             {
  75.                 $parts = split(' ', substr($httpHeaders[0], 9), 2);
  76.  
  77.                 $status = $parts[0];
  78.                 $statusMessage = trim($parts[1]);
  79.  
  80.                 array_shift($httpHeaders);
  81.             }
  82.  
  83.             //Look for the Content-Type response header and determine type
  84.             //and encoding from it (if possible)
  85.             foreach ($httpHeaders as $header)
  86.             {
  87.                 if (substr($header, 0, 13) == 'Content-Type:')
  88.                 {
  89.                     //split content type into two parts if possible
  90.                     $parts = split(';', substr($header, 13), 2);
  91.  
  92.                     $type = trim($parts[0]);
  93.  
  94.                     if ($parts[1])
  95.                     {
  96.                         //split the encoding section again to get the value
  97.                         $parts = split('=', $parts[1], 2);
  98.  
  99.                         if ($parts[1])
  100.                         {
  101.                             $encoding = trim($parts[1]);
  102.                         }
  103.                     }
  104.  
  105.                     break;
  106.                 }
  107.             }
  108.         }
  109.  
  110.         $this->_rawResponse = $rawResponse;
  111.         $this->_type = $type;
  112.         $this->_encoding = $encoding;
  113.         $this->_httpStatus = $status;
  114.         $this->_httpStatusMessage = $statusMessage;
  115.     }

  167.     /**

  175.         if (!$this->_isParsed)

  178.             $this->_isParsed = true;
  179.         }

  181.         if (isset($this->_parsedData->$key))

  189.     /**
  190.      * Parse the raw response into the parsed_data array for access
  191.      */
  192.     private function _parseData()
  193.     {
  194.         //An alternative would be to use Zend_Json::decode(...)
  195.         $data = json_decode($this->_rawResponse);
  196.  
  197.         //convert $data->response->docs[*] to be Solr_Document objects
  198.         if (isset($data->response) && is_array($data->response->docs))
  199.         {
  200.             $documents = array();
  201.  
  202.             foreach ($data->response->docs as $doc)
  203.             {
  204.                 $document = new Apache_Solr_Document();
  205.  
  206.                 foreach ($doc as $key => $value)
  207.                 {
  208.                     //If a result is an array with only a single
  209.                     //value then its nice to be able to access
  210.                     //it as if it were always a single value
  211.                     if (is_array($value) && count($value) <= 1)
  212.                     {
  213.                         $value = array_shift($value);
  214.                     }
  215.  
  216.                     $document->$key = $value;
  217.                 }
  218.  
  219.                 $documents[] = $document;
  220.             }
  221.  
  222.             $data->response->docs = $documents;
  223.         }
  224.  
  225.         //correct facet counts to make sense
  226.         //converts array([field value], [facet count], [field value], [facet count] ...) format
  227.         //to array([field value] => [facet count], ... ) format
  228.         if (isset($data->facet_counts) && isset($data->facet_counts->facet_fields))
  229.         {
  230.             foreach ($data->facet_counts->facet_fields as $key => $facet_array)
  231.             {
  232.                 $new_facet_array = array();
  233.  
  234.                 while (count($facet_array) > 0)
  235.                 {
  236.                     $new_facet_array[array_shift($facet_array)] = array_shift($facet_array);
  237.                 }
  238.  
  239.                 $data->facet_counts->facet_fields->$key = $new_facet_array;
  240.             }
  241.         }
  242.  
  243.         $this->_parsedData = $data;
  244.     }
