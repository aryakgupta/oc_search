  120.     /**
  121.      * Escape a value for special query characters such as ':', '(', ')', '*', '?', etc.
  122.      *
  123.      * NOTE: inside a phrase fewer characters need escaped, use {@link Apache_Solr_Service::escapePhrase()} instead
  124.      *
  125.      * @param string $value 
  126.      * @return string 
  127.      */
  128.     static public function escape($value)
  129.     {
  130.         //list taken from http://lucene.apache.org/java/docs/queryparsersyntax.html#Escaping%20Special%20Characters
  131.         $pattern = '/(\+|-|&&|\|\||!|\(|\)|\{|}|\[|]|\^|"|~|\*|\?|:|\\\)/';
  132.         $replace = '\\\$1';
  133.  
  134.         return preg_replace($pattern, $replace, $value);
  135.     }
  136.

  195.     private function _constructUrl($servlet, $params = array())
  196.     {
  197.         if (count($params))
  198.         {
  199.             //escape all parameters appropriately for inclusion in the query string
  200.             $escapedParams = array();
  201.  
  202.             foreach ($params as $key => $value)
  203.             {
  204.                 $escapedParams[] = urlencode($key) . '=' . urlencode($value);
  205.             }
  206.  
  207.             $queryString = $this->_queryDelimiter . implode($this->_queryStringDelimiter, $escapedParams);

  210.         {
  211.             $queryString = '';
  212.         }
  213.  
  214.         return 'http://' . $this->_host . ':' . $this->_port . $this->_path . $servlet . $queryString;

  397.     /**
  398.      * Call the /admin/ping servlet, can be used to quickly tell if a connection to the
  399.      * server is able to be made.
  400.      *
  401.      * @param float $timeout maximum time to wait for ping in seconds, -1 for unlimited (default is 5)
  402.      * @return float Actual time taken to ping the server, FALSE if timeout occurs
  403.      */
  404.     public function ping($timeout = 5)
  405.     {
  406.         $timeout = (float) $timeout;
  407.  
  408.         if ($timeout <= 0)
  409.         {
  410.             $timeout = -1;
  411.         }
  412.  
  413.         $start = microtime(true);
  414.  
  415.         //try to connect to the host with timeout
  416.         $fp = fsockopen($this->_host, $this->_port, $errno, $errstr, $timeout);
  417.  
  418.         if ($fp)
  419.         {
  420.             //If we have a timeout set, then determine the amount of time we have left
  421.             //in the request and set the stream timeout for the write operation
  422.             if ($timeout > 0)
  423.             {
  424.                 //do the calculation
  425.                 $writeTimeout = $timeout - (microtime(true) - $start);
  426.  
  427.                 //check if we're out of time
  428.                 if ($writeTimeout <= 0)
  429.                 {
  430.                     return false;
  431.                 }
  432.  
  433.                 //convert to microseconds and set the stream timeout
  434.                 $writeTimeoutInMicroseconds = (int) $writeTimeout * 1000000;
  435.                 stream_set_timeout($fp, 0, $writeTimeoutInMicroseconds);
  436.             }
  437.  
  438.             $request =     'HEAD ' . $this->_path . self::PING_SERVLET . ' HTTP/1.1' . "\r\n" .
  439.                         'host: ' . $this->_host . "\r\n" .
  440.                         'Connection: close' . "\r\n" .
  441.                         "\r\n";
  442.  
  443.             fwrite($fp, $request);
  444.  
  445.             //check the stream meta data to see if we timed out during the operation
  446.             $metaData = stream_get_meta_data($fp);
  447.  
  448.             if ($metaData['timeout'])
  449.             {
  450.                 fclose($fp);
  451.                 return false;
  452.             }
  453.  
  454.  
  455.             //if we have a timeout set and have made it this far, determine the amount of time
  456.             //still remaining and set the timeout appropriately before the read operation
  457.             if ($timeout > 0)
  458.             {
  459.                 //do the calculation
  460.                 $readTimeout = $timeout - (microtime(true) - $start);
  461.  
  462.                 //check if we've run out of time
  463.                 if ($readTimeout <= 0)
  464.                 {
  465.                     return false;
  466.                 }
  467.  
  468.                 //convert to microseconds and set the stream timeout
  469.                 $readTimeoutInMicroseconds = $readTimeout * 1000000;
  470.                 stream_set_timeout($fp, 0, $readTimeoutInMicroseconds);
  471.             }
  472.  
  473.             $response = fread($fp, 15);
  474.  
  475.             //check the stream meta data to see if we timed out during the operation
  476.             $metaData = stream_get_meta_data($fp);
  477.  
  478.             if ($metaData['timeout'])
